# terceira mudança
# inicializar um repositório GIT
git init .

# clonar um repositório remoto
git clone <url>

# adicionar endereço remoto em um repositório local
git remote add <url>

# verificar endereço remoto
git remote -v

# enviar alterações e definir a upstream(remoto) com a branch local
git push -u origin <nome_da_branch>

# enviar alterações para o github
git push 
git push origin <nome_da_branch>

# enviar alterações de uma branch local para uma branch remota
git push origin <nome_da_branch_local>:<nome_da_branch_remoto>

# mostrar o estado atual do repositório
git status

# adicionar um arquivo em stage
git add <nome_do_arquivo>

# adicionar todos os arquivos da pasta
git add .

# remover arquivo do stage
git rm --cached <nome_do_arquivo>

# restaurar um arquivo
git restore <nome_do_arquivo>

# salvar o estado do repositório
git commit -m "sua msg"

# add e salvar , commitar os arquivos monitorados(track)
git commit -am "sua msg"

# commitar e substituir o ultimo commit
git commit -m "sua msg" --amend

# mostrar o historico de commits
git log
git log --oneline
git log --oneline -n 5

# mostrar as alterações no repositório
git diff

# exibir alterações de um commit
git show <hash>

# listar as branchs
git branch

# renomear a branch atual
git branch -m <novo_nome>

# criar uma nova branch
git branch <nome_da_branch>

# criar branch e trocar
git checkout -b <nome_da_branch>

# mover o HEAD para um commit
git checkout <hash_do_commit>

# deletar uma branch
git branch -d <nome_da_branch>

# deletar forçadamente uma branch
git branch -D <nome_da_branch>


# trocar de branch
git switch <nome_da_branch>

# mesclar uma branch em outra
git merge <nome_da_branch>

# mostrar autores de um arquivo
git blame <nome_do_arquivo>

# voltar para um commit anterior (LOCAL)
git reset <hash_do_commit>

# desfazer um commit (REMOTO)
git revert <hash_do_commit>

# baixar as atualizações do repositório remoto, sem mesclar
git fetch <nome_da_branch>

# baixar as atualizações do repositório remoto e mesclar
git pull <nome_da_branch>
